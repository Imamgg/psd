    person_path = os.path.join(base_path, person)

        continue

    for action in os.listdir(person_path):
        action_path = os.path.join(person_path, action)
        if not os.path.isdir(action_path):
            continue

        for filename in os.listdir(action_path):
            if filename.endswith(".mp3") or filename.endswith(".wav"):
                file_path = os.path.join(action_path, filename)

                try:
                    signal, sr = librosa.load(file_path, sr=None)
                    features = tsfel.time_series_features_extractor(cfg, signal, verbose=0)

                    features['label_orang'] = person
                    features['label_aksi'] = action
                    features['file'] = filename

                    all_features.append(features)

                    print(f"✅ Selesai ekstraksi: {file_path}")

def predict_voice(audio_path):
    """
    Fungsi untuk memprediksi voice dari file audio baru
    
    Parameters:
    -----------
    audio_path : str
        Path ke file audio (.wav atau .mp3)
    
    Returns:
    --------
    dict : Hasil prediksi dengan informasi detail
    """
    # Load model dan preprocessing objects
    with open('model/model_action.pkl', 'rb') as f:
        model_action = pickle.load(f)
    
    with open('model/model_person.pkl', 'rb') as f:
        model_person = pickle.load(f)
    
    with open('model/scaler_action.pkl', 'rb') as f:
        scaler_action = pickle.load(f)
    
    with open('model/scaler_person.pkl', 'rb') as f:
        scaler_person = pickle.load(f)
    
    with open('model/selected_features_action.pkl', 'rb') as f:
        selected_features_action = pickle.load(f)
    
    with open('model/selected_features_person.pkl', 'rb') as f:
        selected_features_person = pickle.load(f)
    
    with open('model/tsfel_config.pkl', 'rb') as f:
        cfg = pickle.load(f)
    
    # Load dan ekstraksi fitur dari audio
    signal, sr = librosa.load(audio_path, sr=None)
    features = tsfel.time_series_features_extractor(cfg, signal, verbose=0)
    
    # Handle NaN dan Inf
    features = features.replace([np.inf, -np.inf], np.nan)
    features = features.fillna(features.median())
    
    # Prediksi AKSI (buka/tutup)
    X_action = features[selected_features_action]
    X_action_scaled = scaler_action.transform(X_action)
    pred_action = model_action.predict(X_action_scaled)[0]
    pred_action_proba = model_action.predict_proba(X_action_scaled)[0]
    
    # Prediksi ORANG (imam/mufid)
    X_person = features[selected_features_person]
    X_person_scaled = scaler_person.transform(X_person)
    pred_person = model_person.predict(X_person_scaled)[0]
    pred_person_proba = model_person.predict_proba(X_person_scaled)[0]
    
    return {
        'file': audio_path,
        'prediksi_aksi': pred_action,
        'confidence_aksi': max(pred_action_proba) * 100,
        'prediksi_orang': pred_person,
        'confidence_orang': max(pred_person_proba) * 100,
        'probabilitas_aksi': dict(zip(model_action.classes_, pred_action_proba)),
        'probabilitas_orang': dict(zip(model_person.classes_, pred_person_proba))
    }

# Analisis prediksi yang salah
from sklearn.metrics import classification_report

# Prediksi ulang untuk analisis

y_pred_person_full = model_person.predict(X_test_b)

# Confidence scores

y_pred_person_proba = model_person.predict_proba(X_test_b)

# Analisis error untuk AKSI
error_indices_action = np.where(y_pred_action_full != y_test_a)[0]
print("=== Analisis Error: Model AKSI ===")
print(f"Total prediksi salah: {len(error_indices_action)} dari {len(y_test_a)} ({len(error_indices_action)/len(y_test_a)*100:.2f}%)")

if len(error_indices_action) > 0:
    print("\nContoh prediksi yang salah (dengan confidence):")
    for i, idx in enumerate(error_indices_action[:5]):
        true_label = y_test_a.iloc[idx] if hasattr(y_test_a, 'iloc') else y_test_a[idx]
        pred_label = y_pred_action_full[idx]
        confidence = max(y_pred_action_proba[idx]) * 100
        print(f"  {i+1}. True: {true_label}, Predicted: {pred_label}, Confidence: {confidence:.2f}%")
